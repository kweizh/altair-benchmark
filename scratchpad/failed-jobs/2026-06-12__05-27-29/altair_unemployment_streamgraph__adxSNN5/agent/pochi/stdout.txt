    on='pointerover',
    nearest=True,
    encodings=['x'],
    empty=False,

    alt.Chart(source)
    .mark_area(interpolate='monotone')
    .encode(

        y=alt.Y('sum(count):Q', stack='center', axis=None),
        color=alt.Color('series:N', scale=alt.Scale(scheme='category20b')),

    .add_params(hover)
)
