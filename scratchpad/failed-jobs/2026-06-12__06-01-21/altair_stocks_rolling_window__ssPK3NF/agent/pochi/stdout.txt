import altair as alt

url = 'https://cdn.jsdelivr.net/npm/vega-datasets@v1.29.0/data/stocks.csv'

# Create the param for the dropdown
symbol_param = alt.param(
    name='symbol_selector',
    value='MSFT',
    bind=alt.binding_select(
        options=['MSFT', 'AMZN', 'IBM', 'GOOG', 'AAPL'],
        name='Symbol: '
    )
)

base = alt.Chart(url).transform_filter(
    'datum.symbol == symbol_selector'
)

# Layer 1: raw price lines
line1 = base.mark_line(opacity=0.3, strokeWidth=1).encode(
    x='date:T',
    y='price:Q',
    color='symbol:N'
)

# Layer 2: rolling mean
line2 = base.transform_window(
    window=[{'field': 'price', 'op': 'mean', 'as': 'rolling_mean'}],
    frame=[-15, 15],
    groupby=['symbol']
).mark_line(strokeWidth=3).encode(
    x='date:T',
    y='rolling_mean:Q',
    color='symbol:N'
)

# Layer 3: all-time mean rule
line3 = base.transform_joinaggregate(
    joinaggregate=[{'field': 'price', 'op': 'mean', 'as': 'alltime_mean'}],
    groupby=['symbol']
).mark_rule(strokeDash=[5, 5]).encode(
    y='alltime_mean:Q',
    color='symbol:N'
)

chart = alt.layer(line1, line2, line3).add_params(symbol_param).interactive(bind_y=False)

print('Chart spec compiled successfully')
print(chart.to_dict())
