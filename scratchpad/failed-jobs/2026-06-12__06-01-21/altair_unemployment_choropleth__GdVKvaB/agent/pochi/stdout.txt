import altair as alt
from vega_datasets import data

# Quick test to verify the full pipeline works
brush = alt.selection_interval(encodings=['x'], name='brush')

# Map
counties = alt.topo_feature(data.us_10m.url, 'counties')

map_chart = alt.Chart(counties).mark_geoshape().transform_lookup(
    lookup='id',
    from_=alt.LookupData(data=data.unemployment.url, key='id', fields=['rate'])
).encode(
    color=alt.Color('rate:Q', scale=alt.Scale(domain=[0.05, 0.10, 0.15, 0.20], type='threshold', scheme='blues')),
    tooltip=['id:N', 'rate:Q']
).project(
    type='albersUsa'
).transform_filter(
    brush
)

# Histogram
hist = alt.Chart(data.unemployment.url).mark_bar().encode(
    alt.X('rate:Q', bin=True),
    alt.Y('count()')
).add_params(brush)

combined = alt.vconcat(map_chart, hist)
combined.save('/tmp/test_chart.html')
print('Success!')
