import json, re

with open("chart.html") as f:
    html = f.read()

# Pull the Vega-Lite spec JSON out of the vegaEmbed call
m = re.search(r'vegaEmbed\(.*?,\s*(\{.*\})\s*,\s*\{', html, re.DOTALL)
if not m:
    # Try the alt pattern: the spec is stored in a variable
    m = re.search(r'var\s+spec\s*=\s*(\{.*?\});\s*var\s+embedOpt', html, re.DOTALL)
spec_str = m.group(1)
spec = json.loads(spec_str)

print("=== top-level keys:", list(spec.keys()))
print("=== $schema:", spec.get("$schema",""))
print("=== width:", spec.get("width"), " height:", spec.get("height"))
print("=== layer count:", len(spec.get("layer",[])))

for i, layer in enumerate(spec.get("layer", [])):
    mark = layer.get("mark", {})
    if isinstance(mark, str):
        mark = {"type": mark}
    enc = layer.get("encoding", {})
    print(f"\n--- Layer {i} ---")
    print("  mark.type:", mark.get("type"))
    print("  mark.dy  :", mark.get("dy"))
    print("  encoding keys:", list(enc.keys()))
    if "color" in enc:
        print("  color:", json.dumps(enc["color"], indent=4))
    if "opacity" in enc:
        print("  opacity:", json.dumps(enc["opacity"], indent=4))
    if "text" in enc:
        print("  text:", json.dumps(enc["text"], indent=4))
    if "x" in enc:
        print("  x scale:", enc["x"].get("scale"))
    if "y" in enc:
        print("  y scale:", enc["y"].get("scale"))

params = spec.get("params", [])
for layer in spec.get("layer", []):
    params.extend(layer.get("params", []))
print("\n=== params found:", json.dumps(params, indent=2))
