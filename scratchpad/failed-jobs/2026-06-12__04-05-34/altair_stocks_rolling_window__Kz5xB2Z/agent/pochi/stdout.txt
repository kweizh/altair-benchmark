import altair as alt
from vega_datasets import data

dropdown = alt.binding_select(options=['MSFT', 'AMZN', 'IBM', 'GOOG', 'AAPL'], name='Select Symbol: ')
symbol_param = alt.param(
    name='symbol_selection',
    value='MSFT',
    bind=dropdown
)

base = alt.Chart(data.stocks.url).transform_filter(
    alt.datum.symbol == symbol_param
)

raw_line = base.mark_line(opacity=0.3, size=1).encode(
    x='date:T',
    y='price:Q',
    color='symbol:N'
)

rolling_line = base.mark_line(size=3).transform_window(
    rolling_mean='mean(price)',
    frame=[-15, 15],
    groupby=['symbol']
).encode(
    x='date:T',
    y='rolling_mean:Q',
    color='symbol:N'
)

ref_rule = base.mark_rule(strokeDash=[4, 4]).transform_joinaggregate(
    all_time_mean='mean(price)',
    groupby=['symbol']
).encode(
    y='all_time_mean:Q',
    color='symbol:N'
)

chart = alt.layer(raw_line, rolling_line, ref_rule).add_params(symbol_param).interactive(bind_y=False)
print(chart.to_json()[:1000])
