import altair as alt
import pandas as pd
from vega_datasets import data

# Centroids
df_centroids = pd.read_csv('https://cdn.jsdelivr.net/gh/gavinr/world-countries-centroids@v1/dist/countries.csv')
centroid_dict = df_centroids.set_index('COUNTRY')[['longitude', 'latitude']].to_dict('index')
centroid_dict['Hong Kong'] = {'longitude': 114.1747, 'latitude': 22.2783}

df_gap = data.gapminder()
countries_gap = df_gap['country'].unique()
centroids_values = [
    {'country': c, 'longitude': centroid_dict[c]['longitude'], 'latitude': centroid_dict[c]['latitude']}
    for c in countries_gap if c in centroid_dict
]
centroids_data = alt.Data(values=centroids_values)

max_pop = int(df_gap['pop'].max())

# Base layer
countries_base = alt.topo_feature(data.world_110m.url, 'countries')
base = alt.Chart(countries_base).mark_geoshape(
    fill='#eeeeee',
    stroke='white'
)

# Slider
slider = alt.param(
    name='Year',
    value=1955,
    bind=alt.binding_range(min=1955, max=2005, step=5, name='Year')
)

# Bubble layer
bubbles = alt.Chart(data.gapminder.url).mark_circle().transform_lookup(
    lookup='country',
    from_=alt.LookupData(data=centroids_data, key='country', fields=['longitude', 'latitude'])
).transform_filter(
    alt.datum.year == slider
).encode(
    longitude='longitude:Q',
    latitude='latitude:Q',
    size=alt.Size('pop:Q', scale=alt.Scale(domain=[0, max_pop])),
    color='cluster:N'
)

# Layered chart
chart = alt.layer(base, bubbles).project(
    type='naturalEarth1'
).add_params(
    slider
)

print(chart.to_json())
