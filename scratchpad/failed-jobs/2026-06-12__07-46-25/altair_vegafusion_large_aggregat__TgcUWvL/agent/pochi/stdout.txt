import random
import pandas as pd
import altair as alt

random.seed(42)
x = [random.randint(0, 99) for _ in range(1000)]
y = [random.randint(0, 49) for _ in range(1000)]
z = [random.random() for _ in range(1000)]
df = pd.DataFrame({'x': x, 'y': y, 'z': z})

alt.data_transformers.enable('vegafusion')

chart = alt.Chart(df).mark_rect().encode(
    x=alt.X('x:Q', bin=alt.Bin(maxbins=20)),
    y=alt.Y('y:Q', bin=alt.Bin(maxbins=20)),
    color=alt.Color('z:Q', aggregate='mean', scale=alt.Scale(scheme='magma'))
)

td = chart.transformed_data()
print(type(td))
print(td.head())
