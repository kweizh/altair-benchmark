# Pattern: vegaEmbed('#vis', <spec>, ...)

if not m:
    # fallback: look for the spec object directly
    m = re.search(r'var spec\s*=\s*(\{.*?\});', html, re.DOTALL)
if not m:
    print("FAIL: could not find embedded spec")
    exit(1)

spec = json.loads(m.group(1))

# ── helper: flatten nested lists/dicts to find things ────────────────────────
def find_all(obj, key):
    """Recursively yield all values for `key` found anywhere in obj."""
    if isinstance(obj, dict):
        if key in obj:
            yield obj[key]
        for v in obj.values():
            yield from find_all(v, key)
    elif isinstance(obj, list):
        for item in obj:
            yield from find_all(item, key)

def find_all_objects(obj):
    """Yield every dict found anywhere in obj."""
    if isinstance(obj, dict):
        yield obj
        for v in obj.values():
            yield from find_all_objects(v)
    elif isinstance(obj, list):
        for item in obj:
            yield from find_all_objects(item)
